هذا سؤال ممتاز، وأعتبره من أهم الأسئلة التي تحدد جودة النظام.

بعد أن قررنا أن:

```text
كل عميل = حساب

كل مورد = حساب

كل موظف = حساب

كل صندوق = حساب

كل بنك = حساب
```

فهناك قاعدة معمارية يجب اعتمادها في النظام.

وهي:

```text
الكيان التشغيلي (Customer, Supplier, Employee...)
هو المالك لبياناته.

الحساب المحاسبي (Account)
هو انعكاس محاسبي له.
```

أي أن العلاقة ليست متساوية، بل هناك "مصدر حقيقة" (Source of Truth).

وأنا أوصي أن يكون:

```text
Customer
Supplier
Employee
CashBox
Bank
```

هم مصدر الحقيقة للاسم والبيانات العامة.

أما Account فهو يتبعهم.

━━━━━━━━━━━━━━━━━━━━
مثال إنشاء المورد
━━━━━━━━━━━━━━━━━━━━

المستخدم أدخل:

```text
شركة الأمل
```

النظام ينشئ:

```text
Suppliers
---------
Id=5
Name=شركة الأمل
AccountId=2205
```

ثم:

```text
Accounts
---------
Id=2205
Name=شركة الأمل
```

━━━━━━━━━━━━━━━━━━━━
بعد سنة
━━━━━━━━━━━━━━━━━━━━

المورد غيّر اسمه إلى:

```text
شركة الأمل للتجارة
```

المستخدم يدخل شاشة الموردين ويعدل الاسم فقط.

━━━━━━━━━━━━━━━━━━━━
ماذا يجب أن يحدث؟
━━━━━━━━━━━━━━━━━━━━

داخل Transaction واحدة:

1.

تحديث:

```text
Suppliers.Name
```

2.

تحديث:

```text
Accounts.Name
```

3.

حفظ.

━━━━━━━━━━━━━━━━━━━━

وليس:

```text
تحديث المورد فقط.
```

━━━━━━━━━━━━━━━━━━━━
هل يتغير AccountId؟
━━━━━━━━━━━━━━━━━━━━

❌ إطلاقاً.

يتغير فقط:

```text
Account.Name
```

━━━━━━━━━━━━━━━━━━━━
ماذا لا يجب أن يتغير؟
━━━━━━━━━━━━━━━━━━━━

مثال:

```text
Code

Nature

ParentId
```

هذه لا يجب أن تتغير تلقائياً.

━━━━━━━━━━━━━━━━━━━━
إذا غير الهاتف
━━━━━━━━━━━━━━━━━━━━

مثال:

```text
Phone
```

هل نحدث الحساب؟

❌ لا.

الحساب المحاسبي لا يحتوي هاتفاً أصلاً.

━━━━━━━━━━━━━━━━━━━━
إذا غير العنوان
━━━━━━━━━━━━━━━━━━━━

نفس الشيء.

لا يحدث شيء في الحساب.

━━━━━━━━━━━━━━━━━━━━
إذا غير الاسم
━━━━━━━━━━━━━━━━━━━━

✅ يحدث في الحساب.

━━━━━━━━━━━━━━━━━━━━
إذا عطل المورد
━━━━━━━━━━━━━━━━━━━━

```text
IsActive=false
```

هل نعطل الحساب؟

أنا أوصي:

✅ نعم.

━━━━━━━━━━━━━━━━━━━━

حتى لا يظهر في الاختيارات.

━━━━━━━━━━━━━━━━━━━━
إذا أعاد تفعيله
━━━━━━━━━━━━━━━━━━━━

```text
Restore()
```

يعاد تفعيل الحساب أيضاً.

━━━━━━━━━━━━━━━━━━━━
إذا حذف المورد
━━━━━━━━━━━━━━━━━━━━

أنت لا تستخدم Hard Delete.

إذن:

```text
MarkAsDeleted()
```

━━━━━━━━━━━━━━━━━━━━

الحساب أيضاً:

```text
MarkAsDeleted()
```

━━━━━━━━━━━━━━━━━━━━
نفس الفكرة مع العميل
━━━━━━━━━━━━━━━━━━━━

إذا غير:

```text
الاسم
```

يحدث:

```text
Customer.Name

↓

Account.Name
```

━━━━━━━━━━━━━━━━━━━━
الموظف
━━━━━━━━━━━━━━━━━━━━

إذا غير:

```text
محمد

↓

محمد علي
```

━━━━━━━━━━━━━━━━━━━━

الحساب يصبح:

```text
محمد علي
```

━━━━━━━━━━━━━━━━━━━━
الصندوق
━━━━━━━━━━━━━━━━━━━━

إذا أصبح:

```text
الصندوق الرئيسي

↓

صندوق الإدارة
```

━━━━━━━━━━━━━━━━━━━━

الحساب أيضاً:

```text
صندوق الإدارة
```

━━━━━━━━━━━━━━━━━━━━
البنك
━━━━━━━━━━━━━━━━━━━━

إذا تغير:

```text
بنك التضامن

↓

بنك الكريمي
```

━━━━━━━━━━━━━━━━━━━━

الحساب يتغير اسمه أيضاً.

━━━━━━━━━━━━━━━━━━━━
كيف أكتبها في الكود؟
━━━━━━━━━━━━━━━━━━━━

لا أحب أن تكتب:

```csharp
_supplierRepository.Update();

_accountRepository.Update();
```

داخل Service مباشرة.

━━━━━━━━━━━━━━━━━━━━

الأفضل أن يكون عندك Service اسمها مثلاً:

```csharp
IAccountLinkService
```

أو:

```csharp
IAccountSynchronizationService
```

مثال:

```csharp
await _accountSynchronizationService
    .SyncSupplierAsync(supplier);
```

داخلها:

```csharp
Account.Name = Supplier.Name;
```

━━━━━━━━━━━━━━━━━━━━

وبذلك لو تغير المنطق مستقبلاً لن تعدل كل الخدمات.

━━━━━━━━━━━━━━━━━━━━
هل تحتاج Transaction؟
━━━━━━━━━━━━━━━━━━━━

✅ نعم.

لأنك تعدل جدولين:

```text
Suppliers

Accounts
```

في نفس العملية.

━━━━━━━━━━━━━━━━━━━━
القاعدة الذهبية التي أوصي باعتمادها
━━━━━━━━━━━━━━━━━━━━

كل الكيانات المرتبطة بالحسابات تتبع نفس السياسة:

```text
Customer
   │
   └── Account

Supplier
   │
   └── Account

Employee
   │
   └── Account

CashBox
   │
   └── Account

Bank
   │
   └── Account
```

وعند:

- إنشاء.
- تعديل الاسم.
- التعطيل.
- إعادة التفعيل.

يتم مزامنة الحساب تلقائياً داخل نفس Transaction.

━━━━━━━━━━━━━━━━━━━━
أقترح تحسيناً معمارياً إضافياً
━━━━━━━━━━━━━━━━━━━━

بدلاً من أن تحتوي كل Service على منطق إنشاء أو تحديث الحساب، أنشئ خدمة واحدة مثل:

```csharp
IAccountLinkService
```

وتكون مسؤولة عن:

```text
CreateCustomerAccount()

CreateSupplierAccount()

CreateEmployeeAccount()

CreateCashBoxAccount()

CreateBankAccount()

RenameLinkedAccount()

ActivateLinkedAccount()

DeactivateLinkedAccount()
```

وبذلك يصبح لديك مكان واحد فقط يحتوي كل منطق الربط بين الكيانات التشغيلية وشجرة الحسابات، مما يجعل النظام أسهل في الصيانة وأقل عرضة للأخطاء مستقبلاً.
